# scripts/sql-ambiguity-allowlist.txt
#
# Ambiguous-column scanner exceptions — see scripts/src/check-sql-ambiguity.mjs.
#
# Each line is one of:
#   routes/<file>.ts                  → skip the whole file
#   routes/<file>.ts:<col>            → skip only that column within the file
#
# Use sparingly. The scanner exists because a bare column that exists on
# two or more joined relations makes Postgres raise
#   "column reference <col> is ambiguous"
# — a parse-time 500 that shipped to users (the 360° evaluation page).
# The correct fix is almost always to qualify the column with its table
# alias (alias."col"). Only add an entry here for a genuine false
# positive the static scanner can't disambiguate.
#
# Format reminder: paths are relative to artifacts/api-server/src/, so
# the file `artifacts/api-server/src/routes/foo.ts` is `routes/foo.ts`.

# ── Scanner-heuristic FPs (missing-column class) ───────────────────────
# Surfaced once the hardened generic-tolerant extractor began enforcing
# the missing-column check. Each is a false positive of the
# fragment-inlining / alias-resolution heuristic — the real SQL uses the
# correct column:
#   d.deletedAt   → real query is `FROM documents d` (documents HAS deletedAt);
#                   detector mis-binds `d` to a `departments d` in another handler.
#   v.brand       → `v.make AS brand` (output alias; fleet_vehicles has `make`).
#   pr.deductions → `pr."totalDeductions" AS credit` (output alias).
#   grn.status    → `CASE WHEN grn."journalId"… END AS status` (output alias).
#   je.postingDate→ `je."date"::text AS "postingDate"` (output alias).
#   cd.title/type → company_documents is created dynamically at runtime with
#                   title/type cols (INSERT proves it); not in the static dump.
#   ea.isActive   → no literal ea.isActive; alias mis-resolution (employee_assignments has `status`).

routes/employees.ts:v.brand
routes/finance-reports.ts:pr.deductions
