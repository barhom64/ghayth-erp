# scripts/ghost-row-allowlist.txt
#
# Ghost-row scanner exceptions — see scripts/src/check-ghost-rows.mjs.
#
# Each line is one of:
#   routes/<file>.ts                  → skip the whole file
#   routes/<file>.ts:<tableName>      → skip only that table within the file
#
# Use sparingly. The scanner exists because soft-deleted rows leaking
# into normal reads (Task #161) is a high-severity, hard-to-spot bug.
# Only add entries for genuinely intentional cases:
#
#   - Audit / activity log readers that must show deleted records.
#   - "Trash" / restore-from-deleted UIs.
#   - Reports that count or summarise soft-deleted rows.
#
# Format reminder: paths are relative to artifacts/api-server/src/, so
# the file `artifacts/api-server/src/routes/foo.ts` is `routes/foo.ts`.

# cip_costs: the cost-count CTE already filters `"deletedAt" IS NULL`, but the
# scanner flags it because the same statement also references
# construction_in_progress (2 tables) — with 2+ refs it requires an
# alias-qualified predicate, and qualifying it (cip_costs cpc) breaks the
# cipListNPlusOneFix unit test that asserts the exact CTE SQL. The read is
# correctly filtered; this is a scanner false-positive, kept as-is.
routes/finance-algorithms.ts:cip_costs
# chart_of_accounts: balance / subsidiary-balance / transaction-detail queries
# join COA by accountCode (or by id to resolve a subsidiary's own account) for
# coa.type/code/name enrichment. Filtering coa."deletedAt" would null the type
# (dropping a soft-deleted account's posted lines from balance buckets) or drop
# the subsidiary row entirely. Lines are filtered (jl."deletedAt" IS NULL).
# Same enrichment exception as finance-hardening:chart_of_accounts.
routes/finance-algorithms.ts:chart_of_accounts
# store_products: spend-by-product report LEFT JOINs store_products only for
# MAX(p.name) display of historical invoice lines; filtering "deletedAt" would
# blank a since-deleted product's name. Same as finance-purchase:store_products.
routes/finance-algorithms.ts:store_products
# warehouse_categories: inventory-valuation report LEFT JOINs the category only
# for c.name display; filtering "deletedAt" would blank a since-deleted
# category's name while the product (and its stock value) stays. Enrichment-only.
routes/finance-algorithms.ts:warehouse_categories
# chart_of_accounts: cost-center P&L / revenue-expense sums join COA by
# accountCode only to read coa.type for the revenue/expense CASE — same
# balance-enrichment exception as finance-hardening:chart_of_accounts (filtering
# coa."deletedAt" would drop a soft-deleted account's posted lines). The lines
# are filtered (jl."deletedAt" IS NULL).
routes/finance-cost-centers.ts:chart_of_accounts
# cost_centers: this table's soft-delete is STATUS-based (UPDATE ... SET
# status='deleted'); its "deletedAt" column is vestigial (never written — 0 rows
# set it). Every cost_centers read here already filters `status != 'deleted'`,
# so soft-deleted cost centres are correctly excluded. The guard flags it only
# because the column exists; a "deletedAt" IS NULL predicate would be a no-op.
routes/finance-cost-centers.ts:cost_centers
# chart_of_accounts: consolidated balance-sheet / P&L sums join COA by accountCode
# purely to read coa.type for the asset/liability/equity/revenue/expense CASE.
# Adding coa."deletedAt" IS NULL would DROP any posted line whose account was
# later soft-deleted, understating totals and unbalancing the balance sheet
# (assets != liabilities + equity). The lines themselves are filtered
# (jl."deletedAt" IS NULL + je.balancesApplied); the account-delete guard also
# refuses to delete an account that still carries journal lines. Enrichment-only
# exception, same class as the warehouse_products joins below.
routes/finance-hardening.ts:chart_of_accounts
# store_products: enrichment name-lookup for historical PO line items. The
# SELECT only builds an id→name display map; filtering "deletedAt" IS NULL
# would blank a since-deleted product's name on old purchase orders (the
# product was valid when the PO was raised). Same rationale as the
# warehouse_products enrichment exceptions below. Every other table in this
# file stays enforced.
routes/finance-purchase.ts:store_products
# chart_of_accounts: the trial-balance / P&L / financial-statement queries join
# COA by accountCode only to read coa.type (and code/name) for the
# revenue/expense/asset/liability/equity CASE buckets. Filtering coa."deletedAt"
# would null the type for a soft-deleted account and DROP its posted lines from
# the statement totals (understating / unbalancing them). Lines are filtered
# (jl."deletedAt" IS NULL + je.balancesApplied). Same enrichment exception as
# finance-hardening:chart_of_accounts.
routes/finance-reports.ts:chart_of_accounts
# purchase_orders: the supplier paid-total sums join purchase_orders only to
# resolve po."supplierId" for a SUM over supplier_payment_allocations of ACTUAL
# posted payments (spa + je already filter deletedAt + balancesApplied).
# Filtering po."deletedAt" would drop real payments made against a since-deleted
# PO and understate the supplier's paid total — enrichment-only, keep.
routes/finance-reports.ts:purchase_orders
