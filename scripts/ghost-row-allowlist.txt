# scripts/ghost-row-allowlist.txt
#
# Ghost-row scanner exceptions — see scripts/src/check-ghost-rows.mjs.
#
# Each line is one of:
#   routes/<file>.ts                  → skip the whole file
#   routes/<file>.ts:<tableName>      → skip only that table within the file
#
# Use sparingly. The scanner exists because soft-deleted rows leaking
# into normal reads (Task #161) is a high-severity, hard-to-spot bug.
# Only add entries for genuinely intentional cases:
#
#   - Audit / activity log readers that must show deleted records.
#   - "Trash" / restore-from-deleted UIs.
#   - Reports that count or summarise soft-deleted rows.
#
# Format reminder: paths are relative to artifacts/api-server/src/, so
# the file `artifacts/api-server/src/routes/foo.ts` is `routes/foo.ts`.
