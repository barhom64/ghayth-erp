# scripts/workflow-silent-failures-allowlist.txt
#
# Allowlist for `scripts/src/check-workflow-silent-failures.mjs`. One
# entry per line. `#` starts a comment. Stale entries fail the guard.
#
# Two key shapes:
#   <basename>:or-true:<exact trimmed line content>
#   <basename>:continue-on-error:<step id or name>
#
# Add a comment above each entry explaining why the failure is safe to
# ignore. Adding to this file is a code-review red flag — prefer to
# fix the underlying step.

# e2e.yml: failure-diagnostics log tail (`if: failure()` step). `tail` exits 1
# when the log file was never created (a server that failed before writing) —
# tolerating it lets all three tails run so the full diagnostic dump prints.
e2e.yml:or-true:echo "=== api-server.log (tail) ==="; tail -200 /tmp/api-server.log || true
e2e.yml:or-true:echo "=== ghayth-erp.log (tail) ==="; tail -200 /tmp/ghayth-erp.log || true
e2e.yml:or-true:echo "=== proxy.log (tail) ===";       tail -200 /tmp/proxy.log     || true

# e2e.yml: idempotent service teardown in the `if: always()` Stop-services
# block. `kill` is allowed to fail if the PID file is missing (server never
# started) or the PID already exited (graceful shutdown).
e2e.yml:or-true:[ -f "$f" ] && kill "$(cat "$f")" 2>/dev/null || true

# guard.yml: schema_post load uses ON_ERROR_STOP=0 because the dump's
# trailing `\unrestrict` line errors when the session isn't in restricted
# mode (harmless — all DDL above it has already executed). The next
# step is an explicit FK-count assertion (`expected_fks` vs `actual_fks`)
# that DOES exit 1 with both logs attached if the schema isn't fully
# loaded, so the silent failure is structurally re-checked.
guard.yml:or-true:psql "$DATABASE_URL" -v ON_ERROR_STOP=0 -q -f db/schema_post.sql > /tmp/post.log 2>&1 || true
